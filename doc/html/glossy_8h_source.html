327 /* -------------------------- Clock Capture ------------------------- */
333 #define CAPTURE_NEXT_CLOCK_TICK(t_cap_h, t_cap_l) do {\
334  /* Enable capture mode for timers B6 and A2 (ACLK) */\
335  TBCCTL6 = CCIS0 | CM_POS | CAP | SCS; \
336  TACCTL2 = CCIS0 | CM_POS | CAP | SCS; \
337  /* Wait until both timers capture the next clock tick */\
338  while (!((TBCCTL6 & CCIFG) && (TACCTL2 & CCIFG))); \
339  /* Store the capture timer values */\
340  t_cap_h = TBCCR6; \
341  t_cap_l = TACCR2; \
342  /* Disable capture mode */\
343  TBCCTL6 = 0; \
344  TACCTL2 = 0; \
345 } while (0)
